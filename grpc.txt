GRPC
-	Local Project :- C:\workspace\poc\grpc-first-project
-	Need to create one “.proto” file in project.
-	Using this file we can generate server stubs and client. 
-	Gradle plugin will do stubs generation task.
-	Create service and extends Impl class generated by plugin. Generated class will be in build folder.
-	Create one main class which will start server. This application will run using netty libraries.
-	Test newly added grpc service using GRPC client Gui application.
-	To test this I have installed BloomRPC in our system.

------------------
basic build.gradle file =>

plugins {
    id 'java'
    id 'com.google.protobuf' version '0.9.4'
}

group 'org.example'
version '1.0-SNAPSHOT'

repositories {
    mavenCentral()
}

def grpcVersion = '1.62.2' // CURRENT_GRPC_VERSION

dependencies {
    implementation "io.grpc:grpc-protobuf:${grpcVersion}"
    implementation "io.grpc:grpc-services:${grpcVersion}"
    implementation "io.grpc:grpc-stub:${grpcVersion}"

    //Used to start server
    runtimeOnly "io.grpc:grpc-netty-shaded:${grpcVersion}"

}

protobuf {
    protoc {
        artifact = "com.google.protobuf:protoc:3.25.1"
    }
    plugins {
        grpc {
            artifact = 'io.grpc:protoc-gen-grpc-java:1.62.2'
        }
    }
    generateProtoTasks {
        all()*.plugins {
            grpc {}
        }
    }
}
